```tsx
import { Show } from "solid-js";

interface IHomeProps {
  isLoggedIn: boolean;
  firstName: string;
}

function Home(props: IHomeProps) {
  return (
    <Show
      when={props.isLoggedin}
      fallback={
        <>
          <div>Welcome to the application. Please sign in to continue.</div>
          <SignInForm />
        </>
      }
    >
      <div>Welcome back, {props.firstName}!</div>
      <Dashboard />
    </Show>
  );
}
```
