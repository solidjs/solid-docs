const loremIpsumWords = "Lorem ipsum dolor sit amet, consecteturâ€¦".split(/\s+/);
export const createLorem = (words: Accessor<number> | number) => {
  return createMemo(() => {
    const output = [],
      len = typeof words === "function" ? words() : words;
    while (output.length <= len) {
      output.push(...loremIpsumWords);
    }
    return output.slice(0, len).join(" ");
  });
};
