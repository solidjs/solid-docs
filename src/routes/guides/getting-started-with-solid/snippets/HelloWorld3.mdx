```tsx
import { render } from "solid-js/web";

function HelloWorld() {
  const name = "Solid";
  const style = { "background-color": "#2c4f7c", color: "#FFF" };

  return <div style={style}>Hello {name}!</div>;
}

const root = document.getElementById("root");

if (root) {
  render(() => <HelloWorld />, root);
}
```
